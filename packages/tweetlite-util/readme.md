## Install

```
npm install --save tweetlite-util@0.1.0
```

## Usage

```js

import util from 'tweetlite-util';

//or

import {
	notActionHimself,
	okActionLanguage,
	notActionBlocks,
	notActionUrl
} from 'tweetlite-util'; // some blacklist functions

import {
	actionFavorite,
	actionUserFollow,
} from 'twitbot-util'; // some watch middleware functions

import {
	twitbotSettings,
	question,
	prompt,
	spinnerMsg
} from 'tweetlite-util'; // some utility functions


```
